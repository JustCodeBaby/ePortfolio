// Function to add numbers and detect overflow
template 
T add_numbers(T const& start, T const& increment, unsigned long int const& steps) {
    T result = start;
    for (unsigned long int i = 0; i < steps; ++i) {
        try {
            // Perform addition and check for overflow
            T temp = result + increment;
            if ((result > 0 && increment > 0 && temp < 0) ||
                (result < 0 && increment < 0 && temp > 0)) {
                throw std::overflow_error("Overflow detected during addition.");
            }
            result = temp;
        }
        catch (const std::overflow_error& e) {
            // Handle overflow exception and stop the operation
            std::cerr << e.what() << " Last good value: " << result << std::endl;
            break;
        }
    }
    return result;
}

// Function to subtract numbers and detect underflow
template 
T subtract_numbers(T const& start, T const& decrement, unsigned long int const& steps) {
    T result = start;
    for (unsigned long int i = 0; i < steps; ++i) {
        try {
            // Perform subtraction and check for underflow
            T temp = result - decrement;
            if ((result > 0 && decrement > 0 && temp < 0) ||
                (result < 0 && decrement < 0 && temp > 0)) {
                throw std::underflow_error("Underflow detected during subtraction.");
            }
            result = temp;
        }
        catch (const std::underflow_error& e) {
            // Handle underflow exception and stop the operation
            std::cerr << e.what() << " Last good value: " << result << std::endl;
            break;
        }
    }
    return result;
}
